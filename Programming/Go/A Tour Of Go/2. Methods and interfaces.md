## 2.1 –ú–µ—Ç–æ–¥—ã 

–í Go –Ω–µ—Ç –∫–ª–∞—Å—Å–æ–≤. –û–¥–Ω–∞–∫–æ –≤—ã –º–æ–∂–µ—Ç–µ –æ–ø—Ä–µ–¥–µ–ª—è—Ç—å –º–µ—Ç–æ–¥—ã –¥–ª—è —Ç–∏–ø–æ–≤.

–ú–µ—Ç–æ–¥ ‚Äî —ç—Ç–æ —Ñ—É–Ω–∫—Ü–∏—è —Å–æ —Å–ø–µ—Ü–∏–∞–ª—å–Ω—ã–º –∞—Ä–≥—É–º–µ–Ω—Ç–æ–º-–ø–æ–ª—É—á–∞—Ç–µ–ª–µ–º (receiver).

–ü–æ–ª—É—á–∞—Ç–µ–ª—å —É–∫–∞–∑—ã–≤–∞–µ—Ç—Å—è –≤ –æ—Ç–¥–µ–ª—å–Ω–æ–º —Å–ø–∏—Å–∫–µ –∞—Ä–≥—É–º–µ–Ω—Ç–æ–≤ –º–µ–∂–¥—É –∫–ª—é—á–µ–≤—ã–º —Å–ª–æ–≤–æ–º `func` –∏ –∏–º–µ–Ω–µ–º –º–µ—Ç–æ–¥–∞.

–í —ç—Ç–æ–º –ø—Ä–∏–º–µ—Ä–µ –º–µ—Ç–æ–¥ `Abs` –∏–º–µ–µ—Ç –ø–æ–ª—É—á–∞—Ç–µ–ª—è —Ç–∏–ø–∞ `Vertex` —Å –∏–º–µ–Ω–µ–º `v`.

```go
package main

import (
	"fmt"
	"math"
)

type Vertex struct {
	X, Y float64
}

func (v Vertex) Abs() float64 {
	return math.Sqrt(v.X*v.X + v.Y*v.Y)
}

func main() {
	v := Vertex{3, 4}
	fmt.Println(v.Abs())
}
```

**–ú–µ—Ç–æ–¥—ã ‚Äî —ç—Ç–æ —Ñ—É–Ω–∫—Ü–∏–∏** 

–ü–æ–º–Ω–∏—Ç–µ: –º–µ—Ç–æ–¥ ‚Äî —ç—Ç–æ –ø—Ä–æ—Å—Ç–æ —Ñ—É–Ω–∫—Ü–∏—è —Å –∞—Ä–≥—É–º–µ–Ω—Ç–æ–º-–ø–æ–ª—É—á–∞—Ç–µ–ª–µ–º.

–í–æ—Ç `Abs`, –∑–∞–ø–∏—Å–∞–Ω–Ω—ã–π –∫–∞–∫ –æ–±—ã—á–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –±–µ–∑ –∏–∑–º–µ–Ω–µ–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–æ–Ω–∞–ª—å–Ω–æ—Å—Ç–∏.

```go
package main

import (
	"fmt"
	"math"
)

	X, Y float64
}

func Abs(v Vertex) float64 {
	return math.Sqrt(v.X*v.X + v.Y*v.Y)
}

func main() {
	v := Vertex{3, 4}
	fmt.Println(Abs(v))
}
```

–í—ã –º–æ–∂–µ—Ç–µ –æ–±—ä—è–≤–∏—Ç—å –º–µ—Ç–æ–¥ –∏ –¥–ª—è –Ω–µ—Å—Ç—Ä—É–∫—Ç—É—Ä–Ω—ã—Ö —Ç–∏–ø–æ–≤.

–í —ç—Ç–æ–º –ø—Ä–∏–º–µ—Ä–µ –º—ã –≤–∏–¥–∏–º —á–∏—Å–ª–æ–≤–æ–π —Ç–∏–ø `MyFloat` —Å –º–µ—Ç–æ–¥–æ–º `Abs`.

–í—ã –º–æ–∂–µ—Ç–µ –æ–±—ä—è–≤–∏—Ç—å –º–µ—Ç–æ–¥ —Ç–æ–ª—å–∫–æ —Å –ø–æ–ª—É—á–∞—Ç–µ–ª–µ–º, —á–µ–π —Ç–∏–ø –æ–ø—Ä–µ–¥–µ–ª—ë–Ω –≤ —Ç–æ–º –∂–µ –ø–∞–∫–µ—Ç–µ, —á—Ç–æ –∏ —Å–∞–º –º–µ—Ç–æ–¥. –ù–µ–ª—å–∑—è –æ–±—ä—è–≤–∏—Ç—å –º–µ—Ç–æ–¥ —Å –ø–æ–ª—É—á–∞—Ç–µ–ª–µ–º, —á–µ–π —Ç–∏–ø –æ–ø—Ä–µ–¥–µ–ª—ë–Ω –≤ –¥—Ä—É–≥–æ–º –ø–∞–∫–µ—Ç–µ (–≤–∫–ª—é—á–∞—è –≤—Å—Ç—Ä–æ–µ–Ω–Ω—ã–µ —Ç–∏–ø—ã, —Ç–∞–∫–∏–µ –∫–∞–∫ `int`).

```go
package main

import (
	"fmt"
	"math"
)

type MyFloat float64

func (f MyFloat) Abs() float64 {
	if f < 0 {
		return float64(-f)
	}
	return float64(f)
}

	f := MyFloat(-math.Sqrt2)
	fmt.Println(f.Abs())
}
```

**–ü–æ–ª—É—á–∞—Ç–µ–ª–∏-—É–∫–∞–∑–∞—Ç–µ–ª–∏ (Pointer receivers)** 

–í—ã –º–æ–∂–µ—Ç–µ –æ–±—ä—è–≤–ª—è—Ç—å –º–µ—Ç–æ–¥—ã —Å –ø–æ–ª—É—á–∞—Ç–µ–ª—è–º–∏-—É–∫–∞–∑–∞—Ç–µ–ª—è–º–∏.

–≠—Ç–æ –æ–∑–Ω–∞—á–∞–µ—Ç, —á—Ç–æ —Ç–∏–ø –ø–æ–ª—É—á–∞—Ç–µ–ª—è –∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è –∫–∞–∫ `*T` –¥–ª—è –Ω–µ–∫–æ—Ç–æ—Ä–æ–≥–æ —Ç–∏–ø–∞ `T`. (–¢–∞–∫–∂–µ `T` —Å–∞–º –ø–æ —Å–µ–±–µ –Ω–µ –º–æ–∂–µ—Ç –±—ã—Ç—å —É–∫–∞–∑–∞—Ç–µ–ª–µ–º, –Ω–∞–ø—Ä–∏–º–µ—Ä `*int`.)

–ù–∞–ø—Ä–∏–º–µ—Ä, –º–µ—Ç–æ–¥ `Scale` –∑–¥–µ—Å—å –æ–ø—Ä–µ–¥–µ–ª—ë–Ω –¥–ª—è `*Vertex`.

–ú–µ—Ç–æ–¥—ã —Å –ø–æ–ª—É—á–∞—Ç–µ–ª—è–º–∏-—É–∫–∞–∑–∞—Ç–µ–ª—è–º–∏ –º–æ–≥—É—Ç –∏–∑–º–µ–Ω—è—Ç—å –∑–Ω–∞—á–µ–Ω–∏–µ, –Ω–∞ –∫–æ—Ç–æ—Ä–æ–µ —É–∫–∞–∑—ã–≤–∞–µ—Ç –ø–æ–ª—É—á–∞—Ç–µ–ª—å (–∫–∞–∫ –¥–µ–ª–∞–µ—Ç `Scale` –≤ —ç—Ç–æ–º –ø—Ä–∏–º–µ—Ä–µ). –ü–æ—Å–∫–æ–ª—å–∫—É –º–µ—Ç–æ–¥—ã —á–∞—Å—Ç–æ –¥–æ–ª–∂–Ω—ã –∏–∑–º–µ–Ω—è—Ç—å —Å–≤–æ–π –ø–æ–ª—É—á–∞—Ç–µ–ª—å, –ø–æ–ª—É—á–∞—Ç–µ–ª–∏-—É–∫–∞–∑–∞—Ç–µ–ª–∏ –∏—Å–ø–æ–ª—å–∑—É—é—Ç—Å—è —á–∞—â–µ, —á–µ–º –ø–æ–ª—É—á–∞—Ç–µ–ª–∏-–∑–Ω–∞—á–µ–Ω–∏—è.

–ü–æ–ø—Ä–æ–±—É–π—Ç–µ —É–±—Ä–∞—Ç—å `*` –∏–∑ –æ–±—ä—è–≤–ª–µ–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–∏ `Scale` –Ω–∞ —Å—Ç—Ä–æ–∫–µ 16 –∏ –ø–æ—Å–º–æ—Ç—Ä–∏—Ç–µ, –∫–∞–∫ –∏–∑–º–µ–Ω–∏—Ç—Å—è –ø–æ–≤–µ–¥–µ–Ω–∏–µ –ø—Ä–æ–≥—Ä–∞–º–º—ã.

–° –ø–æ–ª—É—á–∞—Ç–µ–ª–µ–º-–∑–Ω–∞—á–µ–Ω–∏–µ–º –º–µ—Ç–æ–¥ `Scale` —Ä–∞–±–æ—Ç–∞–µ—Ç —Å –∫–æ–ø–∏–µ–π –∏—Å—Ö–æ–¥–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è `Vertex`. (–≠—Ç–æ —Ç–∞–∫–æ–µ –∂–µ –ø–æ–≤–µ–¥–µ–Ω–∏–µ, –∫–∞–∫ –∏ —É –ª—é–±–æ–≥–æ –¥—Ä—É–≥–æ–≥–æ –∞—Ä–≥—É–º–µ–Ω—Ç–∞ —Ñ—É–Ω–∫—Ü–∏–∏.) –ú–µ—Ç–æ–¥ `Scale` –¥–æ–ª–∂–µ–Ω –∏–º–µ—Ç—å –ø–æ–ª—É—á–∞—Ç–µ–ª—å-—É–∫–∞–∑–∞—Ç–µ–ª—å, —á—Ç–æ–±—ã –∏–∑–º–µ–Ω–∏—Ç—å –∑–Ω–∞—á–µ–Ω–∏–µ `Vertex`, –æ–±—ä—è–≤–ª–µ–Ω–Ω–æ–µ –≤ —Ñ—É–Ω–∫—Ü–∏–∏ `main`.

```go
package main

import (
	"fmt"
	"math"
)

	X, Y float64
}

	return math.Sqrt(v.X*v.X + v.Y*v.Y)

	v.X = v.X * f
```

**–ú–µ—Ç–æ–¥—ã –∏ –∫–æ—Å–≤–µ–Ω–Ω–∞—è –∞–¥—Ä–µ—Å–∞—Ü–∏—è —É–∫–∞–∑–∞—Ç–µ–ª–µ–π** 

–§—É–Ω–∫—Ü–∏–∏ —Å –∞—Ä–≥—É–º–µ–Ω—Ç–æ–º-—É–∫–∞–∑–∞—Ç–µ–ª–µ–º –¥–æ–ª–∂–Ω—ã –ø—Ä–∏–Ω–∏–º–∞—Ç—å –∏–º–µ–Ω–Ω–æ —É–∫–∞–∑–∞—Ç–µ–ª—å:

```go

ScaleFunc(v, 5)  // –û—à–∏–±–∫–∞ –∫–æ–º–ø–∏–ª—è—Ü–∏–∏!
ScaleFunc(&v, 5) // –î–æ–ø—É—Å—Ç–∏–º–æ
```

–í —Ç–æ –≤—Ä–µ–º—è –∫–∞–∫ –º–µ—Ç–æ–¥—ã —Å –ø–æ–ª—É—á–∞—Ç–µ–ª—è–º–∏-—É–∫–∞–∑–∞—Ç–µ–ª—è–º–∏ –º–æ–≥—É—Ç –≤—ã–∑—ã–≤–∞—Ç—å—Å—è –∫–∞–∫ –Ω–∞ –∑–Ω–∞—á–µ–Ω–∏–∏, —Ç–∞–∫ –∏ –Ω–∞ —É–∫–∞–∑–∞—Ç–µ–ª–µ:

```go
var v Vertex
v.Scale(5)  // –î–æ–ø—É—Å—Ç–∏–º–æ
p := &v
p.Scale(10) // –î–æ–ø—É—Å—Ç–∏–º–æ
```

–í –≤—ã—Ä–∞–∂–µ–Ω–∏–∏ `v.Scale(5)`, —Ö–æ—Ç—è `v` ‚Äî —ç—Ç–æ –∑–Ω–∞—á–µ–Ω–∏–µ, –∞ –Ω–µ —É–∫–∞–∑–∞—Ç–µ–ª—å, –º–µ—Ç–æ–¥ —Å –ø–æ–ª—É—á–∞—Ç–µ–ª–µ–º-—É–∫–∞–∑–∞—Ç–µ–ª–µ–º –≤—ã–∑—ã–≤–∞–µ—Ç—Å—è –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏. –¢–æ –µ—Å—Ç—å, –¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞, Go –∏–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∏—Ä—É–µ—Ç –≤—ã—Ä–∞–∂–µ–Ω–∏–µ `v.Scale(5)` –∫–∞–∫ `(&v).Scale(5)`, –ø–æ—Å–∫–æ–ª—å–∫—É –º–µ—Ç–æ–¥ `Scale` –∏–º–µ–µ—Ç –ø–æ–ª—É—á–∞—Ç–µ–ª—å-—É–∫–∞–∑–∞—Ç–µ–ª—å.

```go
package main

import "fmt"

type Vertex struct {
	X, Y float64
}

func (v *Vertex) Scale(f float64) {
	v.X = v.X * f
	v.Y = v.Y * f
}

func ScaleFunc(v *Vertex, f float64) {
	v.X = v.X * f
	v.Y = v.Y * f
}

func main() {
	v := Vertex{3, 4}
	v.Scale(2)
	ScaleFunc(&v, 10)

	p := &Vertex{4, 3}
	p.Scale(3)
	ScaleFunc(p, 8)

	fmt.Println(v, p)
}
```

**–≠–∫–≤–∏–≤–∞–ª–µ–Ω—Ç–Ω–æ–µ –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç –∏ –≤ –æ–±—Ä–∞—Ç–Ω–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–∏.**

–§—É–Ω–∫—Ü–∏–∏, –ø—Ä–∏–Ω–∏–º–∞—é—â–∏–µ –∞—Ä–≥—É–º–µ–Ω—Ç-–∑–Ω–∞—á–µ–Ω–∏–µ, –¥–æ–ª–∂–Ω—ã –ø—Ä–∏–Ω–∏–º–∞—Ç—å –∑–Ω–∞—á–µ–Ω–∏–µ –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–≥–æ —Ç–∏–ø–∞:

```go

fmt.Println(AbsFunc(&v)) // –û—à–∏–±–∫–∞ –∫–æ–º–ø–∏–ª—è—Ü–∏–∏!
```

–í —Ç–æ –≤—Ä–µ–º—è –∫–∞–∫ –º–µ—Ç–æ–¥—ã —Å –ø–æ–ª—É—á–∞—Ç–µ–ª—è–º–∏-–∑–Ω–∞—á–µ–Ω–∏—è–º–∏ –º–æ–≥—É—Ç –≤—ã–∑—ã–≤–∞—Ç—å—Å—è –∫–∞–∫ –Ω–∞ –∑–Ω–∞—á–µ–Ω–∏–∏, —Ç–∞–∫ –∏ –Ω–∞ —É–∫–∞–∑–∞—Ç–µ–ª–µ:

```go

fmt.Println(v.Abs()) // –î–æ–ø—É—Å—Ç–∏–º–æ

fmt.Println(p.Abs()) // –î–æ–ø—É—Å—Ç–∏–º–æ
```

–í —ç—Ç–æ–º —Å–ª—É—á–∞–µ –≤—ã–∑–æ–≤ –º–µ—Ç–æ–¥–∞ `p.Abs()` –∏–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∏—Ä—É–µ—Ç—Å—è –∫–∞–∫ `(*p).Abs()`.

**–í—ã–±–æ—Ä –ø–æ–ª—É—á–∞—Ç–µ–ª—è-–∑–Ω–∞—á–µ–Ω–∏—è –∏–ª–∏ –ø–æ–ª—É—á–∞—Ç–µ–ª—è-—É–∫–∞–∑–∞—Ç–µ–ª—è** 

–°—É—â–µ—Å—Ç–≤—É–µ—Ç –¥–≤–µ –ø—Ä–∏—á–∏–Ω—ã –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –ø–æ–ª—É—á–∞—Ç–µ–ª—å-—É–∫–∞–∑–∞—Ç–µ–ª—å.

–ü–µ—Ä–≤–∞—è ‚Äî —á—Ç–æ–±—ã –º–µ—Ç–æ–¥ –º–æ–≥ –∏–∑–º–µ–Ω—è—Ç—å –∑–Ω–∞—á–µ–Ω–∏–µ, –Ω–∞ –∫–æ—Ç–æ—Ä–æ–µ —É–∫–∞–∑—ã–≤–∞–µ—Ç –µ–≥–æ –ø–æ–ª—É—á–∞—Ç–µ–ª—å.

–í—Ç–æ—Ä–∞—è ‚Äî —á—Ç–æ–±—ã –∏–∑–±–µ–∂–∞—Ç—å –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è –∑–Ω–∞—á–µ–Ω–∏—è –ø—Ä–∏ –∫–∞–∂–¥–æ–º –≤—ã–∑–æ–≤–µ –º–µ—Ç–æ–¥–∞. –≠—Ç–æ –º–æ–∂–µ—Ç –±—ã—Ç—å –±–æ–ª–µ–µ —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ, –Ω–∞–ø—Ä–∏–º–µ—Ä, –µ—Å–ª–∏ –ø–æ–ª—É—á–∞—Ç–µ–ª—å ‚Äî —ç—Ç–æ –±–æ–ª—å—à–∞—è —Å—Ç—Ä—É–∫—Ç—É—Ä–∞.

–í —ç—Ç–æ–º –ø—Ä–∏–º–µ—Ä–µ –æ–±–∞ –º–µ—Ç–æ–¥–∞ ‚Äî `Scale` –∏ `Abs` ‚Äî –∏–º–µ—é—Ç —Ç–∏–ø –ø–æ–ª—É—á–∞—Ç–µ–ª—è `*Vertex`, —Ö–æ—Ç—è –º–µ—Ç–æ–¥ `Abs` –Ω–µ –æ–±—è–∑–∞–Ω –∏–∑–º–µ–Ω—è—Ç—å —Å–≤–æ–π –ø–æ–ª—É—á–∞—Ç–µ–ª—å.

–í –æ–±—â–µ–º —Å–ª—É—á–∞–µ, –≤—Å–µ –º–µ—Ç–æ–¥—ã –¥–ª—è –æ–¥–Ω–æ–≥–æ —Ç–∏–ø–∞ –¥–æ–ª–∂–Ω—ã –∏–º–µ—Ç—å –ª–∏–±–æ –ø–æ–ª—É—á–∞—Ç–µ–ª–∏-–∑–Ω–∞—á–µ–Ω–∏—è, –ª–∏–±–æ –ø–æ–ª—É—á–∞—Ç–µ–ª–∏-—É–∫–∞–∑–∞—Ç–µ–ª–∏, –Ω–æ –Ω–µ —Å–º–µ—Å—å –æ–±–æ–∏—Ö.

```go
package main

import (
	"fmt"
	"math"
)

type Vertex struct {
	X, Y float64
}

func (v *Vertex) Scale(f float64) {
	v.X = v.X * f
	v.Y = v.Y * f
```

## –ó–Ω–∞—á–µ–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–π 

–§—É–Ω–∫—Ü–∏–∏ ‚Äî —ç—Ç–æ —Ç–æ–∂–µ –∑–Ω–∞—á–µ–Ω–∏—è. –ò—Ö –º–æ–∂–Ω–æ –ø–µ—Ä–µ–¥–∞–≤–∞—Ç—å —Ç–∞–∫ –∂–µ, –∫–∞–∫ –∏ –¥—Ä—É–≥–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è.

–ó–Ω–∞—á–µ–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–π –º–æ–≥—É—Ç –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å—Å—è –∫–∞–∫ –∞—Ä–≥—É–º–µ–Ω—Ç—ã —Ñ—É–Ω–∫—Ü–∏–π –∏ –∫–∞–∫ –≤–æ–∑–≤—Ä–∞—â–∞–µ–º—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è.

```go
package main

import (
	"fmt"
	"math"
)

func compute(fn func(float64, float64) float64) float64 {
	return fn(3, 4)
}

func main() {
	hypot := func(x, y float64) float64 {
		return math.Sqrt(x*x + y*y)
	}
	fmt.Println(hypot(5, 12))

	fmt.Println(compute(hypot))
	fmt.Println(compute(math.Pow))
}
```

**–ó–∞–º—ã–∫–∞–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–π (Function closures)**

## 3.7 –ò–Ω—Ç–µ—Ä—Ñ–µ–π—Å—ã 

–¢–∏–ø –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–∞ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è –∫–∞–∫ –Ω–∞–±–æ—Ä —Å–∏–≥–Ω–∞—Ç—É—Ä –º–µ—Ç–æ–¥–æ–≤.

–ó–Ω–∞—á–µ–Ω–∏–µ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–Ω–æ–≥–æ —Ç–∏–ø–∞ –º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å –ª—é–±–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ, –∫–æ—Ç–æ—Ä–æ–µ —Ä–µ–∞–ª–∏–∑—É–µ—Ç —ç—Ç–∏ –º–µ—Ç–æ–¥—ã.

```go
	"fmt"
	"math"
)

	Abs() float64
}

	var a Abser
	f := MyFloat(-math.Sqrt2)
	v := Vertex{3, 4}

	a = f  // a MyFloat implements Abser
	a = &v // a *Vertex implements Abser

	// In the following line, v is a Vertex (not *Vertex)
	// and does NOT implement Abser.
	// a = v

	fmt.Println(a.Abs())
}

type MyFloat float64

func (f MyFloat) Abs() float64 {
	if f < 0 {
		return float64(-f)
	}
	return float64(f)
}

type Vertex struct {
	X, Y float64
}

func (v *Vertex) Abs() float64 {
	return math.Sqrt(v.X*v.X + v.Y*v.Y)
}
```

**–ü–µ—Ä–µ–∫–ª—é—á–∞—Ç–µ–ª–∏ –ø–æ —Ç–∏–ø—É (Type switches)** 

–ü–µ—Ä–µ–∫–ª—é—á–∞—Ç–µ–ª—å –ø–æ —Ç–∏–ø—É ‚Äî —ç—Ç–æ –∫–æ–Ω—Å—Ç—Ä—É–∫—Ü–∏—è, –ø–æ–∑–≤–æ–ª—è—é—â–∞—è –≤—ã–ø–æ–ª–Ω–∏—Ç—å –Ω–µ—Å–∫–æ–ª—å–∫–æ —É—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–π —Ç–∏–ø–∞ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ.

–ü–µ—Ä–µ–∫–ª—é—á–∞—Ç–µ–ª—å –ø–æ —Ç–∏–ø—É –ø–æ—Ö–æ–∂ –Ω–∞ –æ–±—ã—á–Ω—ã–π –æ–ø–µ—Ä–∞—Ç–æ—Ä `switch`, –Ω–æ –≤ –µ–≥–æ `case`-–≤–µ—Ç–∫–∞—Ö —É–∫–∞–∑—ã–≤–∞—é—Ç—Å—è **—Ç–∏–ø—ã** (–∞ –Ω–µ –∑–Ω–∞—á–µ–Ω–∏—è), –∏ —ç—Ç–∏ —Ç–∏–ø—ã —Å—Ä–∞–≤–Ω–∏–≤–∞—é—Ç—Å—è —Å —Ç–∏–ø–æ–º –∑–Ω–∞—á–µ–Ω–∏—è, —Å–æ–¥–µ—Ä–∂–∞—â–µ–≥–æ—Å—è –≤ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–Ω–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π.

```go
switch v := i.(type) {
case T:
    // –∑–¥–µ—Å—å v –∏–º–µ–µ—Ç —Ç–∏–ø T
case S:
    // –∑–¥–µ—Å—å v –∏–º–µ–µ—Ç —Ç–∏–ø S
default:
    // –Ω–µ—Ç —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π; –∑–¥–µ—Å—å v –∏–º–µ–µ—Ç —Ç–æ—Ç –∂–µ —Ç–∏–ø, —á—Ç–æ –∏ i
}
```

–û–±—ä—è–≤–ª–µ–Ω–∏–µ –≤ –ø–µ—Ä–µ–∫–ª—é—á–∞—Ç–µ–ª–µ –ø–æ —Ç–∏–ø—É –∏–º–µ–µ—Ç —Ç—É –∂–µ —Å–∏–Ω—Ç–∞–∫—Å–∏—á–µ—Å–∫—É—é —Ñ–æ—Ä–º—É, —á—Ç–æ –∏ —É—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ —Ç–∏–ø–∞ `i.(T)`, –Ω–æ –≤–º–µ—Å—Ç–æ –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–≥–æ —Ç–∏–ø–∞ `T` –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –∫–ª—é—á–µ–≤–æ–µ —Å–ª–æ–≤–æ `type`.

–≠—Ç–æ—Ç `switch` –ø—Ä–æ–≤–µ—Ä—è–µ—Ç, —Å–æ–¥–µ—Ä–∂–∏—Ç –ª–∏ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ `i` –∑–Ω–∞—á–µ–Ω–∏–µ —Ç–∏–ø–∞ `T` –∏–ª–∏ `S`. –í –≤–µ—Ç–∫–∞—Ö `case T` –∏ `case S` –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è `v` –±—É–¥–µ—Ç –∏–º–µ—Ç—å —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–π —Ç–∏–ø –∏ –∑–Ω–∞—á–µ–Ω–∏–µ, —Å–æ–¥–µ—Ä–∂–∞—â–µ–µ—Å—è –≤ `i`. –í –≤–µ—Ç–∫–µ `default`, –µ—Å–ª–∏ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π –Ω–µ—Ç, –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è `v` –±—É–¥–µ—Ç –∏–º–µ—Ç—å —Ç–æ—Ç –∂–µ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–Ω—ã–π —Ç–∏–ø –∏ –∑–Ω–∞—á–µ–Ω–∏–µ, —á—Ç–æ –∏ `i`.

```go
package main

import "fmt"

func do(i interface{}) {
	switch v := i.(type) {
	case int:
		fmt.Printf("Twice %v is %v\n", v, v*2)
	case string:
		fmt.Printf("%q is %v bytes long\n", v, len(v))
	default:
		fmt.Printf("I don't know about type %T!\n", v)
	}
}

func main() {
	do(21)
	do("hello")
	do(true)
}
```

**–°—Ç—Ä–æ–∫–æ–≤—ã–µ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å—ã (Stringers)** 

–û–¥–∏–Ω –∏–∑ —Å–∞–º—ã—Ö —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω—ë–Ω–Ω—ã—Ö –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–æ–≤ ‚Äî —ç—Ç–æ `Stringer`, –æ–ø—Ä–µ–¥–µ–ª—ë–Ω–Ω—ã–π –≤ –ø–∞–∫–µ—Ç–µ `fmt`:

–¢–∏–ø `Stringer` ‚Äî —ç—Ç–æ —Ç–∏–ø, –∫–æ—Ç–æ—Ä—ã–π –º–æ–∂–µ—Ç –æ–ø–∏—Å–∞—Ç—å —Å–µ–±—è –≤ –≤–∏–¥–µ —Å—Ç—Ä–æ–∫–∏. –ü–∞–∫–µ—Ç `fmt` (–∏ –º–Ω–æ–≥–∏–µ –¥—Ä—É–≥–∏–µ) –∏—â—É—Ç —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—é —ç—Ç–æ–≥–æ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–∞, —á—Ç–æ–±—ã –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ –≤—ã–≤–æ–¥–∏—Ç—å –∑–Ω–∞—á–µ–Ω–∏—è.

## 3.8 –û—à–∏–±–∫–∏ 

–ü—Ä–æ–≥—Ä–∞–º–º—ã –Ω–∞ Go –≤—ã—Ä–∞–∂–∞—é—Ç —Å–æ—Å—Ç–æ—è–Ω–∏–µ –æ—à–∏–±–∫–∏ —Å –ø–æ–º–æ—â—å—é **–∑–Ω–∞—á–µ–Ω–∏–π –æ—à–∏–±–∫–∏**.

–¢–∏–ø `error` ‚Äî —ç—Ç–æ –≤—Å—Ç—Ä–æ–µ–Ω–Ω—ã–π –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å, –ø–æ—Ö–æ–∂–∏–π –Ω–∞ `fmt.Stringer`:

(–ö–∞–∫ –∏ –≤ —Å–ª—É—á–∞–µ —Å `fmt.Stringer`, –ø–∞–∫–µ—Ç `fmt` –∏—Å–ø–æ–ª—å–∑—É–µ—Ç –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å `error` –ø—Ä–∏ –≤—ã–≤–æ–¥–µ –∑–Ω–∞—á–µ–Ω–∏–π.)

–§—É–Ω–∫—Ü–∏–∏ —á–∞—Å—Ç–æ –≤–æ–∑–≤—Ä–∞—â–∞—é—Ç –∑–Ω–∞—á–µ–Ω–∏–µ —Ç–∏–ø–∞ `error`, –∏ –≤—ã–∑—ã–≤–∞—é—â–∏–π –∫–æ–¥ –¥–æ–ª–∂–µ–Ω –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞—Ç—å –æ—à–∏–±–∫–∏, –ø—Ä–æ–≤–µ—Ä—è—è, —Ä–∞–≤–Ω–∞ –ª–∏ –æ—à–∏–±–∫–∞ `nil`.

–ü—Ä–∏–º–µ—Ä:

```go

    fmt.Printf("couldn't convert number: %v\n", err)
```

–ï—Å–ª–∏ `error == nil`, —ç—Ç–æ –æ–∑–Ω–∞—á–∞–µ—Ç **—É—Å–ø–µ—Ö**. –ï—Å–ª–∏ `error != nil`, —ç—Ç–æ –æ–∑–Ω–∞—á–∞–µ—Ç **–æ—à–∏–±–∫—É**. –¢–∞–∫–æ–π –ø–æ–¥—Ö–æ–¥ –¥–µ–ª–∞–µ—Ç –æ–±—Ä–∞–±–æ—Ç–∫—É –æ—à–∏–±–æ–∫ –≤ Go –ø—Ä–æ—Å—Ç–æ–π –∏ —è–≤–Ω–æ–π.

```go

	"fmt"
	"time"
)

	When time.Time
	What string
}

func (e *MyError) Error() string {
	return fmt.Sprintf("at %v, %s",
		e.When, e.What)
}

func run() error {
	return &MyError{
		time.Now(),
		"it didn't work",
	}
}

	if err := run(); err != nil {
		fmt.Println(err)
	}
}
```

## 3.8 –ß—Ç–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö (Readers) 

–ü–∞–∫–µ—Ç `io` –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å `io.Reader`, –∫–æ—Ç–æ—Ä—ã–π –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ—Ç **–∫–æ–Ω–µ—Ü –ø–æ—Ç–æ–∫–∞ –¥–∞–Ω–Ω—ã—Ö –¥–ª—è —á—Ç–µ–Ω–∏—è**.

–°—Ç–∞–Ω–¥–∞—Ä—Ç–Ω–∞—è –±–∏–±–ª–∏–æ—Ç–µ–∫–∞ Go —Å–æ–¥–µ—Ä–∂–∏—Ç –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Ä–µ–∞–ª–∏–∑–∞—Ü–∏–π —ç—Ç–æ–≥–æ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–∞ ‚Äî –≤–∫–ª—é—á–∞—è **—Ñ–∞–π–ª—ã**, **—Å–µ—Ç–µ–≤—ã–µ —Å–æ–µ–¥–∏–Ω–µ–Ω–∏—è**, **—Å–∂–∞—Ç–∏–µ**, **—à–∏—Ñ—Ä–æ–≤–∞–Ω–∏–µ** –∏ –¥—Ä—É–≥–∏–µ –∏—Å—Ç–æ—á–Ω–∏–∫–∏ –¥–∞–Ω–Ω—ã—Ö.

–ò–Ω—Ç–µ—Ä—Ñ–µ–π—Å `io.Reader` –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç –º–µ—Ç–æ–¥:


–ú–µ—Ç–æ–¥ `Read` –∑–∞–ø–æ–ª–Ω—è–µ—Ç –ø–µ—Ä–µ–¥–∞–Ω–Ω—ã–π —Å—Ä–µ–∑ –±–∞–π—Ç–æ–≤ `b` –¥–∞–Ω–Ω—ã–º–∏ –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç:

–ö–æ–≥–¥–∞ –ø–æ—Ç–æ–∫ –¥–∞–Ω–Ω—ã—Ö –∑–∞–∫–∞–Ω—á–∏–≤–∞–µ—Ç—Å—è, –º–µ—Ç–æ–¥ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –æ—à–∏–±–∫—É `io.EOF`.

üì¶ –ü—Ä–∏–º–µ—Ä: –í –¥–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏–æ–Ω–Ω–æ–º –∫–æ–¥–µ —Å–æ–∑–¥–∞—ë—Ç—Å—è `strings.Reader`, –∏ –µ–≥–æ —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ —á–∏—Ç–∞–µ—Ç—Å—è **–ø–æ 8 –±–∞–π—Ç –∑–∞ —Ä–∞–∑**. –≠—Ç–æ –∏–ª–ª—é—Å—Ç—Ä–∏—Ä—É–µ—Ç, –∫–∞–∫ –º–æ–∂–Ω–æ –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞—Ç—å –ø–æ—Ç–æ–∫–æ–≤—ã–µ –¥–∞–Ω–Ω—ã–µ –ø–æ—Ä—Ü–∏—è–º–∏, —á—Ç–æ –æ—Å–æ–±–µ–Ω–Ω–æ –ø–æ–ª–µ–∑–Ω–æ –ø—Ä–∏ —Ä–∞–±–æ—Ç–µ —Å –±–æ–ª—å—à–∏–º–∏ —Ñ–∞–π–ª–∞–º–∏ –∏–ª–∏ —Å–µ—Ç–µ–≤—ã–º–∏ –ø–æ—Ç–æ–∫–∞–º–∏.

```go
	"fmt"
	"io"
```
